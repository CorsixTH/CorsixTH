105template <typename T>
106class lua_class_binding final {
107 public:
108 lua_class_binding() = delete;
109 lua_class_binding(const lua_class_binding&) = delete;
110 lua_class_binding(lua_class_binding&&) = delete;
111 lua_class_binding& operator=(const lua_class_binding&) = delete;
112 lua_class_binding operator=(lua_class_binding&&) = delete;
113
122 lua_class_binding(const lua_register_state* pState, const char* name,
123 lua_CFunction new_fn, lua_metatable mt)
124 : pState(pState),
125 class_name(name),
126 class_metatable(pState->metatables[static_cast<size_t>(mt)]) {
127 lua_settop(pState->L, pState->top);
128 /* Make metatable the environment for registered functions */
129 lua_pushvalue(pState->L, class_metatable);
130 lua_replace(pState->L, luaT_environindex);
131 /* Set the __gc metamethod to C++ destructor */
132 luaT_pushcclosure(pState->L, luaT_stdgc<T, luaT_environindex>, 0);
133 lua_setfield(pState->L, class_metatable, "__gc");
134 /* Set the depersist size */
135 lua_pushinteger(pState->L, sizeof(T));
136 lua_setfield(pState->L, class_metatable, "__depersist_size");
137 /* Create the methods table; call it -> new instance */
138 luaT_pushcclosuretable(pState->L, new_fn, 0);
139 /* Set __class_name on the methods metatable */
140 lua_getmetatable(pState->L, -1);
141 lua_pushstring(pState->L, class_name);
142 lua_setfield(pState->L, -2, "__class_name");
143 lua_pop(pState->L, 1);
144 /* Set __index to the methods table */
145 lua_pushvalue(pState->L, -1);
146 lua_setfield(pState->L, class_metatable, "__index");
147 }

154 void set_superclass(lua_metatable super_mt) {
155 lua_getmetatable(pState->L, -1);
156 lua_getfield(pState->L, pState->metatables[static_cast<size_t>(super_mt)],
157 "__index");
158 lua_setfield(pState->L, -2, "__index");
159 lua_pop(pState->L, 1);
160 /* Set metatable[1] to super_mt */
161 lua_pushvalue(pState->L, pState->metatables[static_cast<size_t>(super_mt)]);
162 lua_rawseti(pState->L, class_metatable, 1);
163 }
