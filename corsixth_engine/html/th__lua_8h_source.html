59inline void luaT_rotate(lua_State* L, int idx, int n) {
60#if LUA_VERSION_NUM >= 503
61 lua_rotate(L, idx, n);
62#else
63 if (n < 0) {
64 if (idx < 0) {
65 idx = lua_gettop(L) + idx + 1;
66 }
67 n = lua_gettop(L) - idx + 1 + n;
68 }
69
70 for (int i = 0; i < n; i++) {
71 lua_insert(L, idx);
72 }
73#endif
74}

102#if LUA_VERSION_NUM >= 502

373template <class T>
374T* luaT_testuserdata(lua_State* L, int idx, int mt_idx, bool required = true) {
375 // Turn mt_idx into an absolute index, as the stack size changes.
376 if (mt_idx > LUA_REGISTRYINDEX && mt_idx < 0) {
377 mt_idx = lua_gettop(L) + mt_idx + 1;
378 }
379
380 void* ud = lua_touserdata(L, idx);
381 if (ud != nullptr && lua_getmetatable(L, idx) != 0) {
382 while (true) {
383 if (lua_equal(L, mt_idx, -1) != 0) {
384 lua_pop(L, 1);
385 return static_cast<T*>(ud);
386 }
387 // Go up one inheritance level, if there is one.
388 if (lua_type(L, -1) != LUA_TTABLE) break;
389 lua_rawgeti(L, -1, 1);
390 lua_replace(L, -2);
391 }
392 lua_pop(L, 1);
393 }
394
395 if (required) {
396 const char* msg =
397 lua_pushfstring(L, "%s expected, got %s", luaT_classinfo<T>::name(),
398 luaL_typename(L, idx));
399 luaL_argerror(L, idx, msg);
400 assert(false); // unreachable
401 }
402 return nullptr;
403}

468namespace {
469int luaT_get_userdata_classname(lua_State* L, int idx, const char** class_name,
470 void** userdata) {
471 void* p = lua_touserdata(L, idx);
472 *userdata = p;
473 if (p == nullptr) {
474 return 0;
475 }
476 int added_stack = 0;
477 if (lua_getmetatable(L, idx) == 0) {
478 std::printf("Warn: No metatable for userdata\n");
479 return 0;
480 }
481 lua_getfield(L, -1, "__index");
482 added_stack += 2; // metadata and field
483 if (lua_type(L, -1) != LUA_TTABLE) {
484 lua_pop(L, added_stack);
485 std::printf("Warn: No __index field method table for userdata\n");
486 return 0;
487 }
488
489 if (lua_getmetatable(L, -1) == 0) {
490 lua_pop(L, added_stack);
491 std::printf("Warn: No metatable for method table of userdata\n");
492 return 0;
493 }
494 lua_getfield(L, -1, "__class_name");
495 added_stack += 2; // method metatable and class name field
496 if (lua_type(L, -1) != LUA_TSTRING) {
497 lua_pop(L, added_stack);
498 std::printf("Warn: No __class_name field for userdata\n");
499 return 0;
500 }
501 *class_name = lua_tostring(L, -1);
502
503 return added_stack;
504}
505} // namespace

507// Find the appropriate derived class of base for a given
508// userdata object.
509//
510// th_lua_internal.h defines our c++ binding class metatables:
511// metatable: { __index = method_table, ... }
512// method_table: { ... }
513// method_table metatable: { __class_name = class_name }
514template <typename B, typename T1, typename T2>
515B* luaT_touserdata_base(lua_State* L, int idx,
516 const std::initializer_list<const char*>& class_names) {
517 static_assert(std::is_base_of<B, T1>::value, "B must be a base class for T1");
518 static_assert(std::is_base_of<B, T2>::value, "B must be a base class for T2");
519
520 const char* class_name = nullptr;
521 void* p;
522
523 int stack = luaT_get_userdata_classname(L, idx, &class_name, &p);
524 if (class_name == nullptr) {
525 return nullptr;
526 }
527
528 auto it = class_names.begin();
529 if (std::strcmp(class_name, *it) == 0) {
530 lua_pop(L, stack);
531 return static_cast<T1*>(p);
532 }
533 it++;
534 if (std::strcmp(class_name, *it) == 0) {
535 lua_pop(L, stack);
536 return static_cast<T2*>(p);
537 }
538 std::printf("Warn: Unknown class name for usertype %s\n", class_name);
539 lua_pop(L, stack);
540 return nullptr;
541}
