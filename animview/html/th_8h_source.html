23 /*
24  Note that this file contains similar functionality to th.h / th_gfx.h from
25  the main CorsixTH project. It is reimplemented rather than reused for good
26  reasons:
27  1) Animations need to be rendered onto a wxWidgets canvas rather than onto
28  an SDL or DirectX canvas.
29  2) Modifications and experimentation can be performed on this version
30  before being implemented on the game copy, resulting in better code
31  when the game version is implemented.
32  3) Simplicity rather than speed is the aim of this copy, so it will render
33  slower than the game copy, but will be easier to understand and make
34  changes to.
35 */

90 class THLayerMask {
91  public:
92  THLayerMask();
93 
94  inline void set(int iLayer, int iID) {
95  if (0 <= iLayer && iLayer < 13 && 0 <= iID && iID < 32)
96  m_iMask[iLayer] |= (1ul << iID);
97  }
98 
99  void clear();
100 
101  inline void clear(int iLayer, int iID) {
102  if (0 <= iLayer && iLayer < 13 && 0 <= iID && iID < 32)
103  m_iMask[iLayer] &= ~(1ul << iID);
104  }
105 
106  inline bool isSet(int iLayer, int iID) const {
107  if (0 <= iLayer && iLayer < 13 && 0 <= iID && iID < 32)
108  return (m_iMask[iLayer] & (1ul << iID)) != 0;
109  else
110  return false;
111  }
112  inline bool isSet(int iLayer) const {
113  if (0 <= iLayer && iLayer < 13)
114  for (int iId = 0; iId < 32; ++iId) {
115  if ((m_iMask[iLayer] & (1ul << iId)) != 0) return true;
116  }
117  return false;
118  }
119 
120  protected:
121  uint32_t m_iMask[13];
122 };

213  bool loadVector(std::vector<T>& vector, const wxString& sFilename) {
214  vector.clear();
215 
216  wxFile oFile(sFilename);
217  if (!oFile.IsOpened()) return false;
218 
219  size_t iLen = static_cast<size_t>(oFile.Length());
220  unsigned char* pBuffer = new unsigned char[iLen];
221  oFile.Read(pBuffer, iLen);
222  if (memcmp(pBuffer, "RNC\001", 4) == 0) {
223  pBuffer = Decompress(pBuffer, iLen);
224  if (!pBuffer) {
225  return false;
226  }
227  }
228 
229  for (int offset = 0; offset < iLen; offset += sizeof(T)) {
230  vector.push_back(*(reinterpret_cast<T*>(pBuffer + offset)));
231  }
232 
233  return true;
234  }
